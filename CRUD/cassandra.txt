Create Keyspace:
CREATE KEYSPACE library
WITH replication = {'class': 'SimpleStrategy', 'replication_factor': 1};

USE library;

Create Table:
CREATE TABLE books (
    id UUID PRIMARY KEY,
    title TEXT,
    author TEXT,
    published_date DATE
);
